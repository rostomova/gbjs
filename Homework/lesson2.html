  1. Почему код дает именно такие результаты?

    let a = 1, b = 1, c, d;

    c = ++a; alert(c); // 2 - сначала происходит операция +1

    d = b++; alert(d); // 1 - операция +1 происходит после

    c = (2 + ++a); alert(c); // 5 - сначала происходит операция +1 (итого а=3)

    d = (2 + b++); alert(d); // 4 - операция +1 происходит после (итого b=2, а при следующем вызове b=3)

    alert(a); // 3

    alert(b); // 3
